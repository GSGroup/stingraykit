228  bool ReleaseStrongIfUnique()
229  {
230  if (!_value.Get())
231  return false;
232 
233  u32 c = AtomicU32::CompareAndExchange(_value.Get()->_strongReferences, 1, 0);
234  if (c != 1)
235  return false;
236 
237  STINGRAYKIT_ANNOTATE_HAPPENS_AFTER(&_value.Get()->_strongReferences);

243  u32 TryAddStrongReference()
244  {
245  if (!_value.Get())
246  return 0;
247 
248  u32 c = AtomicU32::Load(_value.Get()->_strongReferences);
249  while (c != 0)
250  {
251  u32 newc = AtomicU32::CompareAndExchange(_value.Get()->_strongReferences, c, c + 1);
252  if (newc == c)
253  return newc;
254  c = newc;
255  }
256  return 0;
257  }
