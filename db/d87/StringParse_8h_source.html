123  template < typename Arguments >
124  bool StringParseImpl(const std::string& string, const std::string& format, const Tuple<Arguments>& arguments)
125  {
126  std::deque<variant<TypeList<std::string, size_t>::type > > tokens;
127  std::string::size_type start_pos = 0, current_pos = 0;
128  do
129  {
130  std::string::size_type start_marker_pos = format.find_first_of('%', current_pos);
131  if (start_marker_pos == std::string::npos)
132  break;
133  std::string::size_type end_marker_pos = format.find_first_of('%', start_marker_pos + 1);
134  if (end_marker_pos == std::string::npos)
135  return false;
136 
137  current_pos = end_marker_pos + 1;
138 
139  if (end_marker_pos - start_marker_pos > 1)
140  {
141  std::string substr(format, start_pos, start_marker_pos - start_pos);
142  try
143  {
144  const std::string index_str = std::string(format, start_marker_pos + 1, end_marker_pos - start_marker_pos - 1);
145  size_t index = index_str == "_"? std::numeric_limits<size_t>::max() : FromString<size_t>(index_str);
146  if (!substr.empty())
147  tokens.push_back(substr);
148  tokens.push_back(index);
149  }
150  catch (const std::exception& ex) { continue; }
151  start_pos = current_pos;
152  }
153  }
154  while (current_pos < format.length());
155 
156  if (start_pos < format.length())
157  tokens.push_back(std::string(format, start_pos));
158 
159  size_t index = 0;
160  std::string::size_type current_string_pos = 0;
161  while (!tokens.empty() && current_string_pos < string.length())
162  {
163  if (tokens.front().contains<size_t>())
164  {
165  index = tokens.front().get<size_t>();
166  tokens.pop_front();
167  continue;
168  }
169 
170  std::string substr = variant_get<std::string>(tokens.front());
171  tokens.pop_front();
172  std::string::size_type substr_pos = string.find(substr, current_string_pos);
173  if (substr_pos == std::string::npos)
174  return false;
175 
176  if (index)
177  {
178  if (!(substr_pos - current_string_pos > 0 && ArgumentReader::TryRead(std::string(string, current_string_pos, substr_pos - current_string_pos), arguments, index - 1)))
179  return false;
180  index = 0;
181  }
182  current_string_pos = substr_pos + substr.length();
183  }
184 
185  return tokens.empty() && (index ? ArgumentReader::TryRead(std::string(string.begin() + current_string_pos, string.end()), arguments, index - 1) : !(current_string_pos < string.length()));
186  }
187 
188  }
