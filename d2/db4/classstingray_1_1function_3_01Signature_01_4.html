Detailed Description

template<typename Signature>
class stingray::function< Signature >

Function object.

Example:
double f(char c, int i)
{
return (c - 'a') * i + 0.5;
}
int main()
{
function<double(char, int)> func(&f);
std::cout << func('c', 3) << std::endl; // 9.5
return 0;
}
