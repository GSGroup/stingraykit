72  {

245  virtual bool Process(const Packet<MetadataType>& packet, const ICancellationToken& token)
246  {
247  ConstByteData data(packet.GetData());
248  STINGRAYKIT_CHECK(data.size() <= GetStorageSize(), StringBuilder() % "Packet is too big! Buffer size: " % GetStorageSize() % " packet size:" % data.size());
249 
250  MutexLock l1(_writeMutex); // we need this mutex because write can be called simultaneously from several threads
251  MutexLock l2(_bufferMutex);
252 
253  BithreadCircularBuffer::Writer w = _buffer.Write();
254  size_t padding_size = (w.size() < data.size() && w.IsBufferEnd()) ? w.size() : 0;
255  if (_buffer.GetFreeSize() < padding_size + data.size())
256  {
257  if (_discardOnOverflow)
258  {
259  Logger::Warning() << "Overflow: dropping " << data.size() << " bytes";
260  return true;
261  }
262  else
263  {
264  _bufferFull.Wait(_bufferMutex, token);
265  return false;
266  }
267  }
268 
269  if (padding_size)
270  {
271  _paddingSize = padding_size;
272  w.Push(padding_size);
273  w = _buffer.Write();
274  }
275 
276  {
277  MutexUnlock ul(_bufferMutex);
278  ::memcpy(w.data(), data.data(), data.size());
279  }
280  PacketInfo p(data.size(), packet.GetMetadata());
281  _packetQueue.push_back(p);
282 
283  w.Push(data.size());

314  {
315  r.Pop(_paddingSize);
316  _paddingSize = 0;
317  r = _buffer.Read();
318  }
319

321  STINGRAYKIT_CHECK(p.Size <= r.size(), "Not enough data in packet buffer, need: " + ToString(p.Size) + ", got: " + ToString(r.size()));
322  bool processed = false;
323  {
324  MutexUnlock ul(_bufferMutex);
325  processed = consumer.Process(Packet<MetadataType>(ConstByteData(r.GetData(), 0, p.Size), p.Metadata), token);
326  }
327 
328  if (!processed)
329  return;
330

332  _packetQueue.pop_front();
