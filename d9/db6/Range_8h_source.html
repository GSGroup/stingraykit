98  {

100  typedef RangeBase<IteratorRange<It_>, typename std::iterator_traits<It_>::reference, typename std::iterator_traits<It_>::iterator_category> base;
101 
102  private:
103  const It_ _begin;
104  optional<It_> _it;
105  const It_ _end;
106 
107  public:
108  IteratorRange(const It_& begin, const It_& it, const It_& end) : _begin(begin), _it(it), _end(end)
109  { }
110 
111  bool Valid() const
112  { return _it && *_it != _end; }
113 
114  typename base::ValueType Get()
115  { return **_it; }
116 
117  bool Equals(const IteratorRange& other) const
118  { return _begin == other._begin && _it == other._it && _end == other._end; }
119 
120  Self& First()
121  { _it = _begin; return *this; }
122 
123  Self& Next()
124  {
125  if (!_it)
126  _it = _begin;
127  else
128  ++*_it;
129  return *this;
130  }
131 
132  Self& Last()
133  {
134  if (_begin != _end)
135  {
136  _it = _end;
137  --*_it;
138  }
139  else
140  _it.reset();
141  return *this;
142  }
143 
144  Self& Prev()
145  {
146  if (_it == _begin)
147  _it.reset();
148  else
149  --*_it;
150  return *this;
151  }
152 
153  int GetPosition() const
154  { return _it ? *_it - _begin : -1; }
155 
156  size_t GetSize() const
157  { return _end - _begin; }
158 
159  Self& Move(int distance)
160  {
161  _it = _it ? *_it + distance : _begin + distance - 1;
162  return *this;
163  }
